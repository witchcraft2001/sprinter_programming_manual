# 5.3 Режим 320×256, 8bpp

> **Навигация:** [← 5.2 ZX режим](02_zx_mode.md) | [Оглавление](../README.md) | [5.4 Режим 640×256 →](04_mode2_640.md)

---

## Характеристики

| Параметр | Значение |
|---------|---------|
| Разрешение | 320 × 256 пикселей |
| Биты на пиксель | 8 (256 цветов) |
| Формат | Индексированная палитра 256 цветов из 16 777 216 |
| Размер экрана | 320 × 256 = 81 920 байт |
| Количество страниц 16 КБ | 5 страниц (80 КБ + padding) |
| Double buffering | Да (2 экрана в VRAM) |

Это **основной режим** для нативных программ Sprinter. Он даёт лучшее соотношение: полноцветность + быстрый линейный доступ + работа с акселератором.

---

## Включение режима

### Через DSS API

```asm
        ld      a, 1                ; 1 = 320×256 8bpp
        ld      b, 0                ; видеостраница 0 (экран A)
        ld      c, #50              ; SetVMod
        rst     #10
```

### Через прямые порты

```asm
ALL_MODE        EQU #C3
RGMOD           EQU #C9

        ld      a, 1
        out     (ALL_MODE), a       ; 320×256

        xor     a
        out     (RGMOD), a          ; экран включён, буфер 0
```

## Адресация пикселей

Буфер линейный: **1 байт = 1 пиксель**.

```
Адрес пикселя (x, y) = BASE + y * 320 + x
```

`BASE` — начало VRAM (зависит от того, какая страница подключена в каком окне).

### Простейший вариант: весь экран через окна

80 КБ экрана помещается в 5 последовательных страниц VRAM (`#50`–`#54`). Чтобы иметь произвольный доступ, можно подключить их в разные окна:

```asm
; Подключить первые 4 страницы VRAM в окна для быстрого доступа
        ld      a, #50
        out     (#82), a            ; WIN0: строки 0–51 (16 КБ)
        ld      a, #51
        out     (#A2), a            ; WIN1: строки 51–102
        ld      a, #52
        out     (#C2), a            ; WIN2: строки 102–153
        ld      a, #53
        out     (#E2), a            ; WIN3: строки 153–204
```

Теперь адресация линейна: пиксель `(0,0)` в `0x0000`, `(0,1)` в `0x0140` (320 байт — одна строка), и т.д.

## Вычисление адреса пикселя

```asm
; Вычислить адрес пикселя (X, Y) в буфере
; Вход: DE = X (0..319), HL = Y (0..255)
; Выход: HL = адрес (только в пределах подключённой страницы)
CalcPixelAddr:
        ; HL = Y * 320 = Y * 256 + Y * 64
        ld      h, l
        ld      l, 0                ; HL = Y * 256
        push    hl
        ld      h, 0
        add     hl, hl              ; HL = Y * 2
        add     hl, hl              ; * 4
        add     hl, hl              ; * 8
        add     hl, hl              ; * 16
        add     hl, hl              ; * 32
        add     hl, hl              ; Y * 64
        pop     bc                  ; BC = Y * 256
        add     hl, bc              ; HL = Y * 320
        add     hl, de              ; + X
        ret
```

## Установка пикселя (простой вариант)

```asm
; Нарисовать пиксель цвета B в координатах (X, Y)
; Требует, чтобы нужная страница VRAM была подключена
; Вход: DE = X, HL = Y, B = цвет

SetPixel320:
        ; Определить, в какую страницу VRAM попадает пиксель
        ; Y * 320 = позиция; / 16384 = номер страницы (от начала #50)

        push    bc
        push    de
        ; linear = Y*320 + X
        call    CalcPixelAddr       ; HL = линейный адрес от начала экрана
        ; page_offset = linear / 16384
        ; в окне: adr = (linear mod 16384) | 0xC000

        ld      a, h
        srl     a : srl a           ; A = линейный адрес / 16384 (грубо)
        add     a, #50              ; + базовая страница VRAM
        out     (#E2), a            ; подключить в WIN3

        ld      a, h
        and     #3F                 ; смещение в 16-КБ окне
        or      #C0                 ; привязка к 0xC000
        ld      h, a                ; HL теперь = 0xC000 + offset

        pop     de
        pop     bc
        ld      (hl), b             ; установить пиксель
        ret
```

## Очистка экрана

```asm
; Заполнить весь экран цветом A
ClearScreen320:
        ; Подключаем все 5 страниц VRAM поочерёдно в WIN3
        ld      d, a                ; сохранить цвет
        ld      b, 5                ; 5 страниц
        ld      c, #50              ; начальная страница VRAM
.loop:
        ld      a, c
        out     (#E2), a            ; WIN3 = следующая страница

        ld      hl, #C000
        ld      (hl), d             ; первый байт
        ld      de, #C001
        ld      bc, #3FFF
        ldir                        ; заполнить 16 КБ

        inc     c
        djnz    .loop
        ret
```

## Горизонтальная линия

```asm
; Нарисовать горизонтальную линию длиной L в строке Y, начиная с X
; Вход: HL = Y, DE = X, BC = L, A = цвет
HLine320:
        push    af
        call    CalcPixelAddr       ; HL = начальный адрес
        ; Подключить нужную страницу и сделать FILL через LDIR
        ; (упрощённо)
        pop     af
        ld      (hl), a
        inc     hl
        ld      d, h : ld e, l
        dec     hl
        dec     bc
        ldir
        ret
```

> **Замечание:** для длинных линий и заливок используйте акселератор (`LD C,C`) — это в 4–8 раз быстрее.

## Двойная буферизация

VRAM содержит два экрана:
- **Экран A**: страницы `#50`–`#54`
- **Экран B**: страницы `#55`–`#59`

Порт `#C9` (RGMOD) бит 0 определяет, какой экран отображается. Код **рисует** в неактивный, затем делает flip.

```asm
; Определить, какой экран активен
        in      a, (#C9)
        and     1
        jr      z, .draw_to_b       ; если активен A — рисовать в B

.draw_to_a:
        ; Рисовать в страницы #50-#54
        ld      a, #50
        out     (#E2), a
        jr      .flip

.draw_to_b:
        ; Рисовать в страницы #55-#59
        ld      a, #55
        out     (#E2), a

.flip:
        ; ... рисование ...

        ; Дождаться VSync (через HALT при IM1/IM2)
        halt

        ; Переключить буфер
        in      a, (#C9)
        xor     1
        out     (#C9), a
        ret
```

## Установка палитры

Палитра из 256 записей, каждая — 3 байта (R, G, B в диапазоне 0–63):

```asm
PORT_Y          EQU #89

; Установить цвет index с компонентами R, G, B
SetPaletteEntry:
        ; A = index, C = R, D = G, E = B
        push    af
        ld      a, #C0              ; команда "установить палитру" (см. lib_startup.asm)
        out     (PORT_Y), a
        pop     af
        out     (PORT_Y), a         ; индекс цвета
        ld      a, c
        out     (PORT_Y), a         ; R
        ld      a, d
        out     (PORT_Y), a         ; G
        ld      a, e
        out     (PORT_Y), a         ; B
        ret
```

Через BIOS проще:

```asm
; Через BIOS PIC_SET_PAL
        ld      hl, 5               ; индекс цвета
        ld      e, 63               ; R
        ld      d, 32               ; G
        ld      a, 0                ; B
        ld      c, #A4              ; PIC_SET_PAL
        rst     #08
```

## Пример: рисование простого шаблона

```asm
; Нарисовать градиент 0..255 горизонтальными полосами
DrawGradient:
        ld      b, 0                ; начальная страница VRAM = #50
.page_loop:
        ld      a, b
        add     a, #50
        out     (#E2), a            ; WIN3 = VRAM страница

        ld      hl, #C000
        ld      de, #C001
        ld      a, b
        sla     a                   ; цвет
        ld      (hl), a
        ld      bc, #3FFF
        ldir

        inc     b
        ld      a, b
        cp      5
        jr      c, .page_loop
        ret
```

## Использование акселератора

Для эффективного рисования используйте аппаратный акселератор — см. раздел [5.6 Акселератор](06_accelerator.md).

Пример из `flappybird/src/grx_utils.asm`:
```asm
        di
        ld      d, d                ; LD D,D — установить размер блока
        ld      a, SIZE
        ld      l, l                ; LD L,L — копировать строку
        ei
```

## Ключевые моменты

> - 320×256, 1 байт на пиксель, линейный буфер 81920 байт (5 страниц VRAM).
> - Пиксель (x,y) = `BASE + y*320 + x`.
> - VRAM страницы `#50`–`#54` (экран A), `#55`–`#59` (экран B).
> - Включение: `Dss.SetVMod 1` или `OUT (#C3), 1`.
> - Double buffering через RGMOD бит 0.
> - Палитра: 256 × 3 байта (R, G, B по 0–63), установка через `BIOS PIC_SET_PAL #A4`.
> - Для быстрой заливки и копирования — аппаратный акселератор.
